  c => Math.sin(Math.random() * Math.exp(-c.t*10)),
  async c => {
    let xm1 = 0
    let x = 0
    let y = 0
    return (c, { a = .5 }) => {
      x = +c.input
      y = (y + x - xm1) * a
      xm1 = x
      return y
    }
  }
)
